Server-side JWT flow reference (Node.js)
// 1. Install: npm install docusign-esign jsonwebtoken const docusign = require('docusign-esign'); const jwt = require('jsonwebtoken'); // 2. Build the JWT assertion function buildJwt(integrationKey, userId, rsaKey, isDemo) { const audience = isDemo ? 'account-d.docusign.com' : 'account.docusign.com'; return jwt.sign({ iss: integrationKey, sub: userId, aud: audience, iat: Math.floor(Date.now() / 1000), exp: Math.floor(Date.now() / 1000) + 3600, scope: 'signature impersonation' }, rsaKey, { algorithm: 'RS256' }); } // 3. Exchange JWT for access token async function getAccessToken(jwtToken, isDemo) { const baseUrl = isDemo ? 'https://account-d.docusign.com' : 'https://account.docusign.com'; const res = await fetch(`${baseUrl}/oauth/token`, { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded' }, body: `grant_type=urn:ietf:params:oauth:grant-type:jwt-bearer&assertion=${jwtToken}` }); return res.json(); // { access_token, expires_in, ... } } // 4. Create envelope from template async function createEnvelope(accessToken, accountId, templateId, signer, isDemo) { const client = new docusign.ApiClient(); client.setBasePath(`https://${isDemo ? 'demo' : 'www'}.docusign.net/restapi`); client.addDefaultHeader('Authorization', `Bearer ${accessToken}`); const envelopeDef = { templateId, status: 'sent', templateRoles: [{ email: signer.email, name: signer.name, roleName: 'signer', clientUserId: signer.clientUserId // required for embedded signing }] }; const api = new docusign.EnvelopesApi(client); return api.createEnvelope(accountId, { envelopeDefinition: envelopeDef }); } // 5. Get recipient view (embedded signing URL) async function getSigningUrl(accessToken, accountId, envelopeId, signer, returnUrl, isDemo) { const client = new docusign.ApiClient(); client.setBasePath(`https://${isDemo ? 'demo' : 'www'}.docusign.net/restapi`); client.addDefaultHeader('Authorization', `Bearer ${accessToken}`); const viewRequest = { returnUrl, authenticationMethod: 'none', email: signer.email, userName: signer.name, clientUserId: signer.clientUserId }; const api = new docusign.EnvelopesApi(client); return api.createRecipientView(accountId, envelopeId, { recipientViewRequest: viewRequest }); // returns { url: 'https://demo.docusign.net/Signing/...', ... } }
